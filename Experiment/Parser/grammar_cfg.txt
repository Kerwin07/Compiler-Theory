# LL(1) 文法：支持函数定义 / 类型声明 / 表达式 / 控制流 / 块 / return
# 终结符名须与词法分析器输出的 token type 一致
# 运算符优先级：== != > < >= <= (最低) < + - < * / < ^ < 一元 - (最高)

S -> INT_KW ID LPAREN RPAREN Block
S -> StmtList

Block -> LBRACE StmtList RBRACE

StmtList -> Stmt StmtList
StmtList -> eps

Stmt -> INT_KW ID DeclRest SEMI
Stmt -> ID StmtTail SEMI
Stmt -> IF LPAREN Expr RPAREN Stmt ElseOpt
Stmt -> WHILE LPAREN Expr RPAREN Stmt
Stmt -> LBRACE StmtList RBRACE
Stmt -> RETURN Expr SEMI
Stmt -> SEMI

DeclRest -> ASSIGN Expr
DeclRest -> eps

ElseOpt -> ELSE Stmt
ElseOpt -> eps

StmtTail -> ASSIGN Expr
StmtTail -> LPAREN Args RPAREN
StmtTail -> Term' Comp' Expr'

Expr   -> Comp Expr'
Expr'  -> EQ Comp Expr'
Expr'  -> NE Comp Expr'
Expr'  -> LT Comp Expr'
Expr'  -> GT Comp Expr'
Expr'  -> LE Comp Expr'
Expr'  -> GE Comp Expr'
Expr'  -> eps
Comp   -> Term Comp'
Comp'  -> PLUS Term Comp'
Comp'  -> MINUS Term Comp'
Comp'  -> eps
Term   -> Power Term'
Term'  -> MUL Power Term'
Term'  -> DIV Power Term'
Term'  -> eps
Power  -> Factor PowerRest
PowerRest -> POW Power
PowerRest -> eps
Factor -> LPAREN Expr RPAREN
Factor -> ID FactorRest
Factor -> INT
Factor -> STRING
Factor -> MINUS Factor

FactorRest -> LPAREN Args RPAREN
FactorRest -> eps

Args -> Expr ArgTail
Args -> eps
ArgTail -> COMMA Expr ArgTail
ArgTail -> eps
